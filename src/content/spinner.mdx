## Spinner

A pixel art styled loading spinner component. Features retro styling and smooth animation, perfect for indicating loading states in pixel art themed applications.

### Installation

Add the Spinner component to your project using the shadcn CLI:

```bash
npx shadcn@latest add @pixelact-ui/spinner
```

### Button and Spinner

<CodePreview
  language="tsx"
  code={`import { Spinner } from "@/components/Spinner";
import { Button } from "@/components/ui/pixelact-ui/button";
import React from "react";

export function LoadingButtonWithText() {
  const [isUploading, setIsUploading] = React.useState(false);

  const handleUpload = async () => {
    setIsUploading(true);
    await new Promise(resolve => setTimeout(resolve, 3000));
    setIsUploading(false);
  };

  return (
    <Button onClick={handleUpload} disabled={isUploading} className="flex items-center gap-2">
      {isUploading && (
        <div style={{ height: 16, width: 16 }}>
          <Spinner />
        </div>
      )}
      {isUploading ? "Uploading..." : "Upload File"}
    </Button>
  );
}`}
